    public String orientationAxialTrace;

    public String getClosureOther() {
        return closureOther;
    }

    public void setClosureOther(String closureOther) {
        this.closureOther = closureOther;

        return orientationAxialTrace;
    }

        return vergence;
    }

    public void setVergence(String vergence) {

        return wavelength;

    public void setWavelength(Double wavelength) {
        this.wavelength = wavelength;

        return wavelengthUnit;

        return amplitudeUnit;

        this.amplitudeUnit = amplitudeUnit;

        return foldContinuity;

        return foldContinuityOther;
